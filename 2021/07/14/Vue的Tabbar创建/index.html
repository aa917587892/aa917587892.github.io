Vue的Tabbar创建

Tabbar的页面搭建

总体关系

为了组件的复用性

省时间说直接上代码

tabbar.vue 组件

    <template>
    <div id="tab-bar">
    <slot></slot>
    </div>
</template>
<script>

export default {
    name: "tabbar"
}
</script>
<style >
#tab-bar{
        display: flex;
        background: #efefef;
        position: fixed;
        bottom: 0px;
        left: 0px;
        right: 0px;
        box-shadow: 0px -3px 1px rgba(100, 100, 100, 0.1);

tabbarItem.vue 组件

    <template>
    <div class="tab-bar-item" @click="itemClick">
        <div v-if="!isActive"><slot name="item-icon"></slot></div>
        <div  v-else><slot name="item-icon-active"></slot></div>
        <div  :style="activeColor"><slot name="item-text"></slot></div>
    </div>

</template>
<script>

export default {
    name:'tabbaritem',
    props:{
        path:String,
        clickColor:{
            type:String,
            default:'red'
        }
    },
    data(){
        return{
         
        }
          },
    computed:{
        isActive(){
            return this.$route.path.indexOf(this.path) !==-1  //获取当前不等于this.path的路由的其他另外几个地址 为false 
        },
        activeColor(){
            return this.isActive? {color:this.clickColor}:{}
        }
    },
    methods:{
        itemClick(){
            this.$router.replace(this.path).catch(()=>{})   //获取app.vue中的path:'/地址'，然后进行跳转
            // console.log(this.path);
        }
    }
    }

</script>
<style>

    .tab-bar-item{
        flex: 1;
        text-align: center;
        height: 49px;
        font-size: 14px;
        margin-top:3px;
    }
    .tab-bar-item img{
        width: 24px;
        height: 24px;
        
    }
     .active{
        color: red;
 }
</style>
