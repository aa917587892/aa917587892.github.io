js高阶函数Filter/map/reduce

filter()/map()/reduce三个高阶函数


const aa=[11,23,44,55,120,555] 

//找出所有小于50的数字
 let newnum1=aa.filter(function(n){
     return n < 50;
 })
 //将得到的数字都乘于2
 let newnum2=newnum1.map(function(n){
     return n * 2
 })
 //将得到的数字进行汇总
 let newnum3=newnum2.reduce(function(preValue,n){
     return prevalue + n
 },0)

更加简便的方法

let sum=aa.filter(function(n){
    return n> 50
}).map(function(n){
    return n * 2
}).reduce(function(preValue,n){
    return preValue + n
})

//把他简便成箭头函数
let sum=aa.filter(n => n> 50).map(n => n * 2).reduce((preValue,n) => preValue + n)
